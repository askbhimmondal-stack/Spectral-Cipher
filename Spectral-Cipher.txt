// CORE: IMAGE → PIXEL ARRAY
// =========================================================
function getPixelArray(imageData, mode) {
  const d = imageData.data;
  const pixels = [];
  for (let i = 0; i < d.length; i += 4) {
    if (mode === 'gray') {
      pixels.push((d[i]*0.299 + d[i+1]*0.587 + d[i+2]*0.114) / 255);
    } else if (mode === 'rgb') {
      pixels.push(d[i]/255, d[i+1]/255, d[i+2]/255);
    } else { // rgba
      pixels.push(d[i]/255, d[i+1]/255, d[i+2]/255, d[i+3]/255);
    }
  }
  return pixels;
}

// =========================================================
// CORE: SEEDED PRNG (for key-based scrambling)
// =========================================================
function seededPRNG(seed) {
  let s = seed | 0;
  return () => {
    s = (s * 1664525 + 1013904223) & 0xffffffff;
    return (s >>> 0) / 0xffffffff;
  };
}

async function keyToSeed(keyStr) {
  if (!keyStr) return 42;
  const enc = new TextEncoder().encode(keyStr);
  const hash = await crypto.subtle.digest('SHA-256', enc);
  const arr = new Uint32Array(hash);
  return arr[0];
}

function scrambleArray(arr, seed) {
  const rng = seededPRNG(seed);
  const indices = arr.map((_,i)=>i);
  for (let i = indices.length-1; i > 0; i--) {
    const j = Math.floor(rng() * (i+1));
    [indices[i], indices[j]] = [indices[j], indices[i]];
  }
  const out = new Array(arr.length);
  for (let i = 0; i < arr.length; i++) out[indices[i]] = arr[i];
  return { scrambled: out, indices };
}

function unscrambleArray(arr, seed) {
  const rng = seededPRNG(seed);
  const indices = arr.map((_,i)=>i);
  for (let i = indices.length-1; i > 0; i--) {
    const j = Math.floor(rng() * (i+1));
    [indices[i], indices[j]] = [indices[j], indices[i]];
  }
  const out = new Array(arr.length);
  for (let i = 0; i < arr.length; i++) out[i] = arr[indices[i]];
  return out;
}

// =========================================================
// CORE: PIXEL VALUES → FREQUENCIES → WAV PCM
// =========================================================
function mapPixelToFreq(val, freqMin, freqMax, method) {
  if (method === 'log') {
    return freqMin * Math.pow(freqMax/freqMin, val);
  } else if (method === 'mel') {
    const melMin = 2595 * Math.log10(1 + freqMin/700);
    const melMax = 2595 * Math.log10(1 + freqMax/700);
    const mel = melMin + val * (melMax - melMin);
    return 700 * (Math.pow(10, mel/2595) - 1);
  } else { // linear
    return freqMin + val * (freqMax - freqMin);
  }
}

function mapFreqToPixel(freq, freqMin, freqMax, method) {
  if (method === 'log') {
    return Math.log(freq/freqMin) / Math.log(freqMax/freqMin);
  } else if (method === 'mel') {
    const mel = 2595 * Math.log10(1 + freq/700);
    const melMin = 2595 * Math.log10(1 + freqMin/700);
    const melMax = 2595 * Math.log10(1 + freqMax/700);
    return (mel - melMin) / (melMax - melMin);
  } else {
    return (freq - freqMin) / (freqMax - freqMin);
  }
}

// Build WAV audio signal encoding pixels as frequency modulated sine tones
function buildWavSignal(pixels, sampleRate, freqMin, freqMax, encMethod, strength) {
  // Each pixel is encoded as one "tone frame"
  // Frame duration: shorter for more pixels, minimum 1ms
  const frameSize = Math.max(2, Math.floor(sampleRate * strength * 0.001));
  const totalSamples = pixels.length * frameSize;
  const pcm = new Float32Array(totalSamples);

  for (let pi = 0; pi < pixels.length; pi++) {
    const freq = mapPixelToFreq(Math.max(0,Math.min(1,pixels[pi])), freqMin, freqMax, encMethod);
    const offset = pi * frameSize;
    for (let si = 0; si < frameSize; si++) {
      const t = (offset + si) / sampleRate;
      pcm[offset + si] = Math.sin(2 * Math.PI * freq * t) * 0.7;
    }
  }
  return { pcm, frameSize };
}

// Build WAV ArrayBuffer
function buildWavFile(pcm, sampleRate, bits) {
  const numSamples = pcm.length;
  const bytesPerSample = bits === 8 ? 1 : bits === 16 ? 2 : 4;
  const dataSize = numSamples * bytesPerSample;
  const buf = new ArrayBuffer(44 + dataSize);
  const view = new DataView(buf);

  function writeStr(offset, str) {
    for (let i = 0; i < str.length; i++) view.setUint8(offset+i, str.charCodeAt(i));
  }
  writeStr(0, 'RIFF');
  view.setUint32(4, 36 + dataSize, true);
  writeStr(8, 'WAVE');
  writeStr(12, 'fmt ');
  view.setUint32(16, 16, true);
  view.setUint16(20, bits === 32 ? 3 : 1, true); // PCM or IEEE float
  view.setUint16(22, 1, true); // mono
  view.setUint32(24, sampleRate, true);
  view.setUint32(28, sampleRate * bytesPerSample, true);
  view.setUint16(32, bytesPerSample, true);
  view.setUint16(34, bits, true);
  writeStr(36, 'data');
  view.setUint32(40, dataSize, true);

  let offset = 44;
  for (let i = 0; i < numSamples; i++) {
    const v = Math.max(-1, Math.min(1, pcm[i]));
    if (bits === 8) {
      view.setUint8(offset, Math.round((v + 1) * 127.5));
      offset += 1;
    } else if (bits === 16) {
      view.setInt16(offset, Math.round(v * 32767), true);
      offset += 2;
    } else {
      view.setFloat32(offset, v, true);
      offset += 4;
    }
  }
  return buf;
}

// Parse WAV → Float32Array PCM
function parseWav(buf) {
  const view = new DataView(buf);
  const audioFormat = view.getUint16(20, true);
  const numChannels = view.getUint16(22, true);
  const sampleRate = view.getUint32(24, true);
  const bitsPerSample = view.getUint16(34, true);
  const dataSize = view.getUint32(40, true);
  const numSamples = dataSize / (bitsPerSample / 8);
  const pcm = new Float32Array(numSamples);
  let offset = 44;
  for (let i = 0; i < numSamples; i++) {
    if (bitsPerSample === 8) {
      pcm[i] = (view.getUint8(offset) / 127.5) - 1;
      offset += 1;
    } else if (bitsPerSample === 16) {
      pcm[i] = view.getInt16(offset, true) / 32767;
      offset += 2;
    } else {
      pcm[i] = view.getFloat32(offset, true);
      offset += 4;
    }
  }
  return { pcm, sampleRate, bitsPerSample, numChannels };
}

// =========================================================
// CORE: DECODE WAV → PIXELS
// =========================================================
// Use zero-crossing + instantaneous frequency estimation
function decodeWavToPixels(pcm, frameSize, pixelCount, sampleRate, freqMin, freqMax, encMethod) {
  const pixels = [];
  for (let pi = 0; pi < pixelCount; pi++) {
    const offset = pi * frameSize;
    // Estimate frequency via autocorrelation on the frame
    const frame = pcm.slice(offset, offset + frameSize);
    const freq = estimateFrequency(frame, sampleRate);
    const val = mapFreqToPixel(freq, freqMin, freqMax, encMethod);
    pixels.push(Math.max(0, Math.min(1, val)));
  }
  return pixels;
}

function estimateFrequency(frame, sampleRate) {
  // Autocorrelation-based pitch detection
  const n = frame.length;
  if (n < 4) return 440;
  let bestLag = -1, bestCorr = -Infinity;
  const minLag = Math.floor(sampleRate / 22050);
  const maxLag = Math.floor(n / 2);
  for (let lag = minLag; lag < maxLag; lag++) {
    let corr = 0;
    for (let i = 0; i < n - lag; i++) corr += frame[i] * frame[i + lag];
    if (corr > bestCorr) { bestCorr = corr; bestLag = lag; }
  }
  if (bestLag <= 0) return 440;
  return sampleRate / bestLag;
}

// =========================================================
// SPECTROGRAM DRAWING
// =========================================================
function drawSpectrogram(cvs, pcm, sampleRate) {
  const ctx = cvs.getContext('2d');
  const W = cvs.width = cvs.parentElement.clientWidth || 800;
  const H = cvs.height;

  const fftSize = parseInt(document.getElementById('setFFTSize')?.value || 1024);
  const hopSize = Math.floor(fftSize / 4);
  const numFrames = Math.floor((pcm.length - fftSize) / hopSize);
  if (numFrames <= 0) return;

  const freqBins = fftSize / 2;
  const img = ctx.createImageData(W, H);
  const hann = new Float32Array(fftSize).map((_,i) => 0.5*(1-Math.cos(2*Math.PI*i/fftSize)));

  // Colors: dark→green→cyan→white (cyber palette)
  function heatColor(v) {
    v = Math.max(0, Math.min(1, v));
    if (v < 0.25) return [0, Math.round(v*4*80), Math.round(v*4*40)];
    if (v < 0.5)  return [0, 80+Math.round((v-0.25)*4*175), 40+Math.round((v-0.25)*4*215)];
    if (v < 0.75) return [Math.round((v-0.5)*4*255), 255, 255];
    return [255, 255, Math.round(255*(1-(v-0.75)*4*0.5))];
  }

  for (let fx = 0; fx < W; fx++) {
    const frameIdx = Math.floor(fx / W * numFrames);
    const start = frameIdx * hopSize;
    const windowed = new Float32Array(fftSize);
    for (let i = 0; i < fftSize; i++) windowed[i] = (pcm[start+i]||0) * hann[i];
    const spectrum = fftMagnitude(windowed);

    for (let fy = 0; fy < H; fy++) {
      const bin = Math.floor((1 - fy/H) * freqBins);
      const mag = Math.log10(1 + spectrum[bin] * 100) / 2;
      const [r,g,b] = heatColor(mag);
      const idx = (fy * W + fx) * 4;
      img.data[idx]=r; img.data[idx+1]=g; img.data[idx+2]=b; img.data[idx+3]=255;
    }
  }
  ctx.putImageData(img, 0, 0);
}

function drawWaveform(cvs, pcm) {
  const ctx = cvs.getContext('2d');
  const W = cvs.width = cvs.parentElement.clientWidth || 800;
  const H = cvs.height;
  ctx.clearRect(0,0,W,H);
  ctx.fillStyle = '#000';
  ctx.fillRect(0,0,W,H);
  ctx.strokeStyle = 'rgba(0,255,180,0.8)';
  ctx.lineWidth = 1;
  ctx.beginPath();
  const step = Math.ceil(pcm.length / W);
  for (let x = 0; x < W; x++) {
    let max = -1, min = 1;
    for (let i = x*step; i < (x+1)*step && i < pcm.length; i++) {
      if (pcm[i]>max) max=pcm[i]; if (pcm[i]<min) min=pcm[i];
    }
    const y1 = (1-max)/2*H, y2 = (1-min)/2*H;
    if (x===0) ctx.moveTo(x, (y1+y2)/2); else { ctx.lineTo(x,y1); ctx.lineTo(x,y2); }
  }
  ctx.stroke();
}

// Simple DFT magnitude (real FFT approximation for visualization)
function fftMagnitude(data) {
  const n = data.length;
  const half = n >> 1;
  const re = new Float32Array(n);
  const im = new Float32Array(n);
  for (let i = 0; i < n; i++) re[i] = data[i];
  // Cooley-Tukey FFT
  fft(re, im, n);
  const mag = new Float32Array(half);
  for (let i = 0; i < half; i++) mag[i] = Math.sqrt(re[i]*re[i]+im[i]*im[i]);
  return mag;
}

function fft(re, im, n) {
  for (let i = 1, j = 0; i < n; i++) {
    let bit = n >> 1;
    for (; j & bit; bit >>= 1) j ^= bit;
    j ^= bit;
    if (i < j) { [re[i],re[j]]=[re[j],re[i]]; [im[i],im[j]]=[im[j],im[i]]; }
  }
  for (let len = 2; len <= n; len <<= 1) {
    const ang = -2*Math.PI/len;
    const wRe = Math.cos(ang), wIm = Math.sin(ang);
    for (let i = 0; i < n; i += len) {
      let curRe=1, curIm=0;
      for (let j = 0; j < len/2; j++) {
        const uRe=re[i+j], uIm=im[i+j];
        const vRe=re[i+j+len/2]*curRe-im[i+j+len/2]*curIm;
        const vIm=re[i+j+len/2]*curIm+im[i+j+len/2]*curRe;
        re[i+j]=uRe+vRe; im[i+j]=uIm+vIm;
        re[i+j+len/2]=uRe-vRe; im[i+j+len/2]=uIm-vIm;
        const nr=curRe*wRe-curIm*wIm;
        curIm=curRe*wIm+curIm*wRe; curRe=nr;
      }
    }
  }
}

// MAIN ENCRYPT FLOW
// =========================================================
async function runEncrypt() {
  if (!state.srcImageData) { notify('No image loaded', true); return; }

  const mode = document.getElementById('colorMode').value;
  const freqRangeStr = document.getElementById('freqRange').value;
  const [fMin, fMax] = freqRangeStr.split('-').map(Number);
  const strength = parseInt(document.getElementById('encStrength').value);
  const keyStr = document.getElementById('encKey').value || document.getElementById('setDefKey').value;
  const scramble = document.getElementById('scrambleMode').checked;
  const stegoMode = document.getElementById('stegoMode').checked;
  const sampleRate = parseInt(document.getElementById('setSampleRate')?.value || 44100);
  const bits = parseInt(document.getElementById('setBits')?.value || 16);
  const encMethod = document.getElementById('setEncMethod')?.value || 'log';

  const encLog = 'encLog';
  const bar = document.getElementById('encProgress');
  bar.style.width = '0%';

  log(encLog, `Starting encryption [${mode.toUpperCase()} mode]`, 'info');
  log(encLog, `Image: ${state.srcWidth}×${state.srcHeight}`, '');
  log(encLog, `Freq range: ${fMin}–${fMax} Hz`, '');

  // Step 1: Extract pixels
  bar.style.width = '10%';
  await tick();
  let pixels = getPixelArray(state.srcImageData, mode);
  log(encLog, `Extracted ${pixels.length} pixel values`, 'ok');

  // Step 2: Scramble
  bar.style.width = '25%';
  await tick();
  let seed = 42;
  if (keyStr) {
    seed = await keyToSeed(keyStr);
    log(encLog, `Key hash seed: 0x${seed.toString(16).toUpperCase()}`, 'info');
  }

  if (scramble) {
    const { scrambled } = scrambleArray(pixels, seed);
    pixels = scrambled;
    log(encLog, 'Pixel scrambling applied', 'ok');
  }

  // Step 3: Build PCM signal
  bar.style.width = '45%';
  await tick();
  const { pcm, frameSize } = buildWavSignal(pixels, sampleRate, fMin, fMax, encMethod, strength);
  log(encLog, `PCM signal built: ${pcm.length} samples @ ${sampleRate} Hz`, 'ok');

  // Step 4: Stego mode — mix with noise
  let finalPcm = pcm;
  if (stegoMode && document.getElementById('setAddNoise').checked) {
    finalPcm = new Float32Array(pcm.length);
    for (let i = 0; i < pcm.length; i++) {
      finalPcm[i] = pcm[i] * 0.35 + (Math.random()*2-1) * 0.65;
    }
    log(encLog, 'STEGO: Signal embedded in noise cover', 'warn');
  }

  // Step 5: Build WAV
  bar.style.width = '65%';
  await tick();
  state.encWavBuffer = buildWavFile(finalPcm, sampleRate, bits);
  log(encLog, `WAV file built: ${(state.encWavBuffer.byteLength/1024).toFixed(1)} KB`, 'ok');

  // Step 6: Metadata
  const metaHash = await hashBuffer(state.encWavBuffer);
  state.encMetadata = {
    version: '2.0.0',
    timestamp: new Date().toISOString(),
    width: state.srcWidth, height: state.srcHeight,
    colorMode: mode, freqMin: fMin, freqMax: fMax,
    frameSize, sampleRate, bits, encMethod,
    scramble, stegoMode,
    pixelCount: pixels.length,
    hash: metaHash,
  };
  log(encLog, `Metadata hash: ${metaHash.slice(0,16)}…`, 'info');

  // Step 7: Draw spectrogram
  bar.style.width = '85%';
  await tick();
  document.getElementById('encResultPanel').style.display = 'block';
  drawSpectrogram(document.getElementById('specCanvas'), finalPcm, sampleRate);
  drawWaveform(document.getElementById('waveCanvas'), finalPcm);
  log(encLog, 'Spectrogram visualization rendered', 'ok');

  bar.style.width = '100%';
  document.getElementById('btnDlWav').style.display = 'inline-flex';
  document.getElementById('btnDlSpec').style.display = 'inline-flex';
  document.getElementById('btnDlJson').style.display = 'inline-flex';
  document.getElementById('btnPlayWav').disabled = false;

  // Store original for comparison
  state.originalForCompare = state.srcImageData;

  log(encLog, 'ENCRYPTION COMPLETE', 'ok');
  notify('Image encrypted successfully!');
}

// =========================================================
// MAIN DECRYPT FLOW
// =========================================================
async function runDecrypt() {
  if (!state.decWavBuffer) { notify('No WAV file loaded', true); return; }

  const keyStr = document.getElementById('decKey').value || document.getElementById('setDefKey').value;
  const decLog = 'decLog';
  const bar = document.getElementById('decProgress');
  bar.style.width = '0%';

  log(decLog, 'Starting decryption…', 'info');

  // Parse WAV
  bar.style.width = '15%';
  await tick();
  let wavData;
  try { wavData = parseWav(state.decWavBuffer); }
  catch(e) { log(decLog, 'ERROR: Invalid WAV file', 'err'); notify('Invalid WAV file', true); return; }

  log(decLog, `WAV parsed: ${wavData.pcm.length} samples @ ${wavData.sampleRate} Hz`, 'ok');

  // Get metadata — try from last encryption, otherwise prompt user
  let meta = state.encMetadata;
  if (!meta) {
    // Try to infer from WAV length; use current settings as fallback
    const fRange = document.getElementById('freqRange').value;
    const [fMin, fMax] = fRange.split('-').map(Number);
    // Need to estimate dimensions — ask user or use WAV properties
    log(decLog, 'No metadata file found. Using current settings.', 'warn');
    // We'll try a square image
    const sampleRate = wavData.sampleRate;
    const strength = parseInt(document.getElementById('encStrength').value);
    const frameSize = Math.max(2, Math.floor(sampleRate * strength * 0.001));
    const pixelCount = Math.floor(wavData.pcm.length / frameSize);
    const side = Math.round(Math.sqrt(pixelCount));
    meta = {
      width: side, height: side, colorMode: 'gray',
      freqMin: fMin, freqMax: fMax, frameSize,
      sampleRate, encMethod: document.getElementById('setEncMethod')?.value || 'log',
      scramble: document.getElementById('scrambleMode').checked,
      pixelCount: side*side,
    };
    log(decLog, `Inferred: ${side}×${side} grayscale`, 'warn');
  }

  bar.style.width = '30%';
  await tick();

  // Hash check
  if (document.getElementById('setVerifyHash')?.checked && meta.hash) {
    const currentHash = await hashBuffer(state.decWavBuffer);
    if (currentHash === meta.hash) log(decLog, 'Hash integrity: VERIFIED ✓', 'ok');
    else log(decLog, 'Hash mismatch — file may be corrupted or tampered', 'warn');
  }

  // Decode pixels
  bar.style.width = '55%';
  await tick();
  let pixels = decodeWavToPixels(
    wavData.pcm, meta.frameSize, meta.pixelCount,
    meta.sampleRate, meta.freqMin, meta.freqMax, meta.encMethod
  );
  log(decLog, `Decoded ${pixels.length} pixel values`, 'ok');

  // Unscramble
  if (meta.scramble) {
    let seed = 42;
    if (keyStr) seed = await keyToSeed(keyStr);
    pixels = unscrambleArray(pixels, seed);
    log(decLog, 'Pixel unscrambling applied', 'ok');
  }

  bar.style.width = '75%';
  await tick();

  // Reconstruct image
  const W = meta.width, H = meta.height;
  const mode = meta.colorMode;
  const cvs = document.createElement('canvas');
  cvs.width = W; cvs.height = H;
  const ctx = cvs.getContext('2d');
  const imgData = ctx.createImageData(W, H);

  const channelsPerPixel = mode === 'gray' ? 1 : mode === 'rgb' ? 3 : 4;
  for (let i = 0; i < W*H; i++) {
    const base = i * 4;
    if (mode === 'gray') {
      const v = Math.round(pixels[i]*255);
      imgData.data[base]=v; imgData.data[base+1]=v; imgData.data[base+2]=v; imgData.data[base+3]=255;
    } else if (mode === 'rgb') {
      imgData.data[base]=Math.round((pixels[i*3]||0)*255);
      imgData.data[base+1]=Math.round((pixels[i*3+1]||0)*255);
      imgData.data[base+2]=Math.round((pixels[i*3+2]||0)*255);
      imgData.data[base+3]=255;
    } else {
      imgData.data[base]=Math.round((pixels[i*4]||0)*255);
      imgData.data[base+1]=Math.round((pixels[i*4+1]||0)*255);
      imgData.data[base+2]=Math.round((pixels[i*4+2]||0)*255);
      imgData.data[base+3]=Math.round((pixels[i*4+3]||1)*255);
    }
  }
  ctx.putImageData(imgData, 0, 0);
  state.reconImageData = imgData;
  state.reconCanvas = cvs;

  bar.style.width = '90%';

  // Show preview
  const decBox = document.getElementById('decPreviewBox');
  decBox.innerHTML = '';
  const img = new Image();
  img.onload = () => { decBox.appendChild(img); };
  img.src = cvs.toDataURL();
  img.style.maxWidth = '100%';
  img.style.maxHeight = '300px';

  document.getElementById('decStats').style.display = 'flex';
  document.getElementById('decStatW').textContent = W;
  document.getElementById('decStatH').textContent = H;
  document.getElementById('decStatCh').textContent = mode.toUpperCase();
  document.getElementById('decStatFid').textContent = '~HIGH';

  document.getElementById('btnDlRecon').style.display = 'inline-flex';

  // Comparison panel
  if (state.originalForCompare) {
    setupComparison(state.originalForCompare, imgData, W, H);
    document.getElementById('comparePanel').style.display = 'block';
  }

  bar.style.width = '100%';
  log(decLog, 'IMAGE RECONSTRUCTION COMPLETE', 'ok');
  notify('Wave decrypted — image reconstructed!');
}

// COMPARISON SLIDER
// =========================================================
function setupComparison(origData, reconData, W, H) {
  const cvs = document.getElementById('compareCanvas');
  const container = document.getElementById('compareContainer');
  const divider = document.getElementById('compareDivider');
  cvs.width = W; cvs.height = H;
  cvs.style.width = '100%';
  state.comparePos = 0.5;

  function drawCompare() {
    const ctx = cvs.getContext('2d');
    const splitX = Math.round(W * state.comparePos);

    // Draw original left
    const leftData = ctx.createImageData(W, H);
    leftData.data.set(origData.data);
    ctx.putImageData(leftData, 0, 0);

    // Draw recon right
    const rightData = ctx.createImageData(W, H);
    rightData.data.set(reconData.data);
    const tmpCvs = document.createElement('canvas');
    tmpCvs.width=W; tmpCvs.height=H;
    tmpCvs.getContext('2d').putImageData(rightData, 0, 0);
    ctx.drawImage(tmpCvs, splitX, 0, W-splitX, H, splitX, 0, W-splitX, H);

    // Divider line
    ctx.strokeStyle = '#00ffb4';
    ctx.lineWidth = 2;
    ctx.beginPath(); ctx.moveTo(splitX,0); ctx.lineTo(splitX,H); ctx.stroke();

    const pct = (state.comparePos*100).toFixed(0);
    divider.style.left = pct+'%';
  }

  drawCompare();

  let dragging = false;
  container.onmousedown = () => dragging = true;
  document.onmouseup = () => dragging = false;
  document.onmousemove = e => {
    if (!dragging) return;
    const rect = container.getBoundingClientRect();
    state.comparePos = Math.max(0, Math.min(1, (e.clientX-rect.left)/rect.width));
    drawCompare();
  };
  container.ontouchstart = e => { dragging = true; };
  document.ontouchend = () => dragging = false;
  document.ontouchmove = e => {
    if (!dragging) return;
    const rect = container.getBoundingClientRect();
    state.comparePos = Math.max(0, Math.min(1, (e.touches[0].clientX-rect.left)/rect.width));
    drawCompare();
  };
}

async function startMicViz() {
  if (state.micStream) return;
  try {
    state.micStream = await navigator.mediaDevices.getUserMedia({ audio: true });
    state.micCtx = new AudioContext();
    const src = state.micCtx.createMediaStreamSource(state.micStream);
    state.micAnalyser = state.micCtx.createAnalyser();
    state.micAnalyser.fftSize = 2048;
    src.connect(state.micAnalyser);
    document.getElementById('micStatus').textContent = 'MIC: LIVE';
    document.getElementById('micStatus').style.color = 'var(--neon)';

    // Rolling spectrogram state
    const specWidth = 800, specHeight = 200;
    const specImgBuf = new Uint8ClampedArray(specWidth * specHeight * 4).fill(0);
    let specCol = 0;

    function drawMic() {
      state.micRaf = requestAnimationFrame(drawMic);
      const bufLen = state.micAnalyser.frequencyBinCount;
      const freqData = new Uint8Array(bufLen);
      const timeData = new Uint8Array(state.micAnalyser.fftSize);
      state.micAnalyser.getByteFrequencyData(freqData);
      state.micAnalyser.getByteTimeDomainData(timeData);

      // Rolling spectrogram
      const lSpec = document.getElementById('liveSpecCanvas');
      const W2 = lSpec.width = lSpec.parentElement.clientWidth || specWidth;
      const H2 = lSpec.height;
      const sCtx = lSpec.getContext('2d');
      const existing = sCtx.getImageData(0,0,W2,H2);
      const shifted = sCtx.createImageData(W2,H2);
      // Shift left by 1
      for (let y = 0; y < H2; y++) {
        for (let x = 0; x < W2-1; x++) {
          const si=(y*W2+x+1)*4, di=(y*W2+x)*4;
          shifted.data[di]=existing.data[si]; shifted.data[di+1]=existing.data[si+1];
          shifted.data[di+2]=existing.data[si+2]; shifted.data[di+3]=255;
        }
      }
      sCtx.putImageData(shifted,0,0);
      // Draw new column on right
      for (let y = 0; y < H2; y++) {
        const bin = Math.floor((1-y/H2)*bufLen);
        const v = freqData[bin]/255;
        const r=Math.round(v*100), g=Math.round(50+v*205), b=Math.round(v*180);
        sCtx.fillStyle=`rgb(${r},${g},${b})`;
        sCtx.fillRect(W2-1,y,1,1);
      }

      // Freq bars
      const fCvs = document.getElementById('freqCanvas');
      const fW = fCvs.width = fCvs.parentElement.clientWidth||800;
      const fH = fCvs.height;
      const fCtx = fCvs.getContext('2d');
      fCtx.clearRect(0,0,fW,fH);
      fCtx.fillStyle='#000';
      fCtx.fillRect(0,0,fW,fH);
      const barW = fW/128;
      for (let i=0;i<128;i++) {
        const v=freqData[i]/255;
        const h=v*fH;
        const hue = 160 - v*130;
        fCtx.fillStyle=`hsl(${hue},100%,${40+v*30}%)`;
        fCtx.fillRect(i*barW, fH-h, barW-1, h);
      }

      // Waveform
      const wCvs = document.getElementById('liveWaveCanvas');
      const wW = wCvs.width = wCvs.parentElement.clientWidth||800;
      const wH = wCvs.height;
      const wCtx = wCvs.getContext('2d');
      wCtx.clearRect(0,0,wW,wH);
      wCtx.fillStyle='#000'; wCtx.fillRect(0,0,wW,wH);
      wCtx.strokeStyle='rgba(0,255,180,0.8)'; wCtx.lineWidth=1.5;
      wCtx.beginPath();
      const sliceW = wW/timeData.length;
      for (let i=0;i<timeData.length;i++) {
        const v=timeData[i]/128-1;
        const x=i*sliceW, y=(1-v)/2*wH;
        i===0?wCtx.moveTo(x,y):wCtx.lineTo(x,y);
      }
      wCtx.stroke();
    }
    drawMic();
  } catch(e) {
    notify('Microphone access denied', true);
  }
}

  if (state.micRaf) cancelAnimationFrame(state.micRaf);
  if (state.micStream) { state.micStream.getTracks().forEach(t=>t.stop()); state.micStream=null; }
  if (state.micCtx) { state.micCtx.close(); state.micCtx=null; }
  document.getElementById('micStatus').textContent = 'MIC: OFFLINE';
  document.getElementById('micStatus').style.color = '';
}

async function playEncryptedWav() {
  if (!state.encWavBuffer) { notify('No encrypted wave file', true); return; }
  if (state.wavAudioCtx) { state.wavAudioCtx.close(); state.wavAudioCtx=null; }
  state.wavAudioCtx = new AudioContext();
  try {
    const decoded = await state.wavAudioCtx.decodeAudioData(state.encWavBuffer.slice(0));
    state.wavSource = state.wavAudioCtx.createBufferSource();
    state.wavSource.buffer = decoded;

    // Analyser for live viz
    const analyser = state.wavAudioCtx.createAnalyser();
    analyser.fftSize = 1024;
    state.wavSource.connect(analyser);
    analyser.connect(state.wavAudioCtx.destination);
    state.wavSource.start();
    document.getElementById('btnStopWav').disabled = false;

    // Visualize on playbackSpec
    function drawPlayback() {
      if (!state.wavAudioCtx) return;
      requestAnimationFrame(drawPlayback);
      const cvs = document.getElementById('playbackSpec');
      const W = cvs.width = cvs.parentElement.clientWidth||800;
      const H = cvs.height;
      const ctx = cvs.getContext('2d');
      const freqData = new Uint8Array(analyser.frequencyBinCount);
      analyser.getByteFrequencyData(freqData);
      const existing = ctx.getImageData(0,0,W,H);
      const shifted = ctx.createImageData(W,H);
      for (let y=0;y<H;y++) for (let x=0;x<W-1;x++) {
        const si=(y*W+x+1)*4, di=(y*W+x)*4;
        shifted.data[di]=existing.data[si]; shifted.data[di+1]=existing.data[si+1];
        shifted.data[di+2]=existing.data[si+2]; shifted.data[di+3]=255;
      }
      ctx.putImageData(shifted,0,0);
      for (let y=0;y<H;y++) {
        const bin=Math.floor((1-y/H)*analyser.frequencyBinCount);
        const v=freqData[bin]/255;
        ctx.fillStyle=`rgb(0,${Math.round(v*255)},${Math.round(v*180)})`;
        ctx.fillRect(W-1,y,1,1);
      }
    }
    drawPlayback();
    notify('Playing encrypted wave…');
  } catch(e) { notify('Playback error', true); }
}

// HASH UTILITY
// =========================================================
async function hashBuffer(buf) {
  const hashBuf = await crypto.subtle.digest('SHA-256', buf);
  return Array.from(new Uint8Array(hashBuf)).map(b=>b.toString(16).padStart(2,'0')).join('');
}
